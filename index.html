how to install on designdobrasil.co
1
Paste the script — Copy the script below and add it once inside the <head> tag of your site's HTML.
2
Tag each portfolio link — Add data-ddb-portfolio="Name" to every <a> link you want to track. The name must match exactly what you added in the Portfolios tab.
3
Open this tracker — Every click on your site gets saved. Open this page to see the stats anytime.
script — paste inside <head> on your site
<script> (function(){ function track(){ document.querySelectorAll('a[data-ddb-portfolio]').forEach(function(el){ if(el._ddb) return; el._ddb = true; el.addEventListener('click', function(){ var name = el.getAttribute('data-ddb-portfolio'); try { var data = JSON.parse(localStorage.getItem('ddb_clicks')||'{"clicks":[]}'); data.clicks.push({ name: name, ts: Date.now() }); localStorage.setItem('ddb_clicks', JSON.stringify(data)); } catch(e){} }); }); } document.addEventListener('DOMContentLoaded', track); new MutationObserver(track).observe(document.documentElement,{childList:true,subtree:true}); })(); </script>
